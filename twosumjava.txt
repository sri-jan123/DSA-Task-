class Solution {

    //iterate through the array elemnts and check that if the sum equals the target or not.Use a temp array to store the target and return it
    public int[] twoSum(int[] nums, int target) {
        int temp[]=new int[2];
        for(int i=0;i<nums.length;i++){
            for(int j=i+1;j<nums.length;j++){
                if(nums[i]+nums[j]==target){
                 temp[0]=i;
                 temp[1]=j;
                }
            }
        }
        return temp;
    }
}
